1//! Source code location attachment collector.
2//!
3//! This module provides functionality to automatically capture and attach
4//! source code location information (file, line, column) to reports when they
5//! are created.
6
7use rootcause_internals::handlers::{AttachmentFormattingStyle, AttachmentHandler};
8
9use crate::hooks::report_creation::AttachmentCollectorHook;
10
11/// Source code location information.
12///
13/// Represents the file, line, and column where a report was created.
14/// This information is automatically captured using
15/// [`core::panic::Location::caller()`].
16#[derive(Copy, Clone, Debug)]
17pub struct Location {
18    /// The source file path where the report was created.
19    pub file: &'static str,
20    /// The line number where the report was created.
21    pub line: u32,
22}
23
24/// Handler for formatting [`Location`] attachments.
25///
26/// This handler formats location information as `filename:line` for both
27/// [`Display`] and [`Debug`] formatting.
28///
29/// [`Display`]: core::fmt::Display
30/// [`Debug`]: core::fmt::Debug
31#[derive(Copy, Clone)]
32pub struct LocationHandler;
33impl AttachmentHandler<Location> for LocationHandler {
34    fn display(value: &Location, formatter: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
35        write!(formatter, "{}:{}", value.file, value.line)
36    }
37
38    fn debug(value: &Location, formatter: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
39        Self::display(value, formatter)
40    }
41
42    fn preferred_formatting_style(
43        _value: &Location,
44        _report_formatting_function: rootcause_internals::handlers::FormattingFunction,
45    ) -> AttachmentFormattingStyle {
46        AttachmentFormattingStyle {
47            priority: 20,
48            ..Default::default()
49        }
50    }
51}
52
53/// Attachment collector for capturing source code location information.
54///
55/// When registered as a report creation hook, this collector automatically
56/// captures the source location where each report is created and attaches
57/// it as a [`Location`] attachment.
58///
59/// ## Example
60///
61/// ```rust
62/// use rootcause::hooks::{
63///     builtin_hooks::location::LocationCollector,
64///     report_creation::register_attachment_collector_hook,
65/// };
66///
67/// // Register to automatically collect location for all reports
68/// register_attachment_collector_hook(LocationCollector);
69/// ```
70#[derive(Copy, Clone)]
71pub struct LocationCollector;
72
73impl AttachmentCollectorHook<Location> for LocationCollector {
74    type Handler = LocationHandler;
75
76    fn collect(&self) -> Location {
77        let location = core::panic::Location::caller();
78        Location {
79            file: location.file(),
80            line: location.line(),
81        }
82    }
83}
