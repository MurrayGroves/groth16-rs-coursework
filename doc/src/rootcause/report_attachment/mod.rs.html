1//! Individual attachments for error reports.
2//!
3//! This module provides types for creating and working with individual
4//! attachments that can be added to error reports. Attachments allow you to
5//! include additional context, data, or information alongside the main error
6//! message.
7//!
8//! # Core Types
9//!
10//! - [`ReportAttachment`]: An owned attachment that can be added to a report
11//! - [`ReportAttachmentRef`]: A reference to an attachment, typically obtained
12//!   from a report
13//!
14//! # Creating Attachments
15//!
16//! Attachments can be created using the [`ReportAttachment::new`] method or
17//! the [`report_attachment!`] macro:
18//!
19//! ```
20//! use rootcause::{prelude::*, report_attachment::ReportAttachment};
21//!
22//! // Create a simple string attachment
23//! let attachment: ReportAttachment<&str> = ReportAttachment::new("Additional information");
24//! let attachment: ReportAttachment<dyn Any> = report_attachment!("Additional information");
25//!
26//! // Create an attachment from a custom type using Debug formatting
27//! #[derive(Debug, Clone)]
28//! struct AttachmentData {
29//!     file: String,
30//!     line: u32,
31//! }
32//!
33//! let attachment_data = AttachmentData {
34//!     file: "main.rs".to_string(),
35//!     line: 42,
36//! };
37//! let debug_attachment: ReportAttachment<AttachmentData> =
38//!     ReportAttachment::new_custom::<handlers::Debug>(attachment_data.clone());
39//! let debug_attachment: ReportAttachment<AttachmentData> = report_attachment!(attachment_data);
40//! ```
41//!
42//! # Type Parameters
43//!
44//! Both types have the same generic parameters:
45//!
46//! - **Attachment type**: Can be a concrete type or `dyn Any` for type erasure
47//! - **Thread safety**: [`SendSync`] (default) for thread-safe attachments, or
48//!   [`Local`] for single-threaded use
49//!
50//! ```
51//! use std::rc::Rc;
52//!
53//! use rootcause::{markers, prelude::*, report_attachment::ReportAttachment};
54//!
55//! // Send + Sync attachment (default)
56//! let attachment: ReportAttachment<String, markers::SendSync> =
57//!     ReportAttachment::new("Thread-safe data".to_string());
58//!
59//! // Local-only attachment (cannot be sent between threads)
60//! let local_data = Rc::new("Local data".to_string());
61//! let local_attachment: ReportAttachment<Rc<String>, markers::Local> =
62//!     ReportAttachment::new(local_data);
63//!
64//! // Send + Sync attachments can be converted to local-only
65//! let local_conversion: ReportAttachment<String, markers::Local> = attachment.into_local();
66//! ```
67//!
68//! # Attachment References
69//!
70//! When attachments are stored in reports, you typically work with
71//! [`ReportAttachmentRef`] which provides access to the attachment data without
72//! taking ownership:
73//!
74//! ```
75//! use rootcause::{prelude::*, report_attachment::ReportAttachment};
76//!
77//! let attachment = ReportAttachment::new("Important context");
78//! let mut report = report!("An error occurred");
79//! report.attachments_mut().push(attachment.into_dyn_any());
80//!
81//! // Access the attachment through a reference
82//! let attachment_ref = report.attachments().get(0).unwrap();
83//! println!("Attachment: {}", attachment_ref);
84//! ```
85//!
86//! # Type Erasure and Downcasting
87//!
88//! Attachments support type erasure through `dyn Any`, allowing collections of
89//! different attachment types. You can downcast back to concrete types when
90//! needed:
91//!
92//! ```
93//! use std::any::{Any, TypeId};
94//!
95//! use rootcause::{prelude::*, report_attachment::ReportAttachment};
96//!
97//! let attachment: ReportAttachment<&str> = ReportAttachment::new("text data");
98//! let erased: ReportAttachment<dyn Any> = attachment.into_dyn_any();
99//!
100//! // Check the type at runtime
101//! assert_eq!(erased.inner_type_id(), TypeId::of::<&str>());
102//!
103//! // Safely downcast back to the original type
104//! let attachment_ref = erased.as_ref();
105//! let typed_ref = attachment_ref.downcast_attachment::<&str>().unwrap();
106//! ```
107//!
108//! # Formatting and Handlers
109//!
110//! Attachments use handlers to control how they are formatted in reports.
111//! The default handler uses the [`Display`] handler, but you can also use
112//! the [`Debug`] handler or create your own:
113//!
114//! ```
115//! # use core::any::Any;
116//! use rootcause::{prelude::*, report_attachment::ReportAttachment};
117//!
118//! #[derive(Debug)]
119//! struct MyData {
120//!     value: i32,
121//! }
122//!
123//! // Use Display formatting (default)
124//! let display_attachment: ReportAttachment<&str> = ReportAttachment::new("text");
125//!
126//! // Use Debug formatting explicitly
127//! let debug_attachment: ReportAttachment<MyData> =
128//!     ReportAttachment::new_custom::<handlers::Debug>(MyData { value: 42 });
129//!
130//! struct MyOtherData {
131//!     value: i32,
132//! }
133//!
134//! struct MyDataHandler;
135//! impl rootcause::handlers::AttachmentHandler<MyOtherData> for MyDataHandler {
136//!     fn debug(data: &MyOtherData, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
137//!         write!(f, "MyOtherData value is: {}", data.value)
138//!     }
139//!
140//!     fn display(data: &MyOtherData, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
141//!         write!(f, "MyOtherData value is: {}", data.value)
142//!     }
143//! }
144//!
145//! let custom_attachment: ReportAttachment<MyOtherData> =
146//!     ReportAttachment::new_custom::<MyDataHandler>(MyOtherData { value: 100 });
147//! ```
148//!
149//! [`SendSync`]: crate::markers::SendSync
150//! [`Local`]: crate::markers::Local
151//! [`Display`]: crate::handlers::Display
152//! [`Debug`]: crate::handlers::Debug
