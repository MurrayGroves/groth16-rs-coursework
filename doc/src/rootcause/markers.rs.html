1//! Marker types and traits for defining ownership and thread-safety semantics.
2//!
3//! This module provides type-level markers that control how reports and
4//! attachments behave with respect to ownership, cloning, and thread safety.
5//! These markers are used as generic parameters in types like [`Report<C, O,
6//! T>`](crate::Report) to encode compile-time guarantees about how data can be
7//! accessed and shared.
8//!
9//! # Design Philosophy
10//!
11//! The constraints encoded by these markers are enforced at construction time.
12//! It is impossible to construct a [`Report`](crate::Report),
13//! [`ReportRef`], or
14//! [`ReportCollection`] that violates the invariants
15//! associated with its marker types. This means you can trust that a
16//! `Report<_, _, SendSync>` truly is `Send + Sync`, and that a `Report<_,
17//! Mutable>` truly has unique ownership.
18//!
19//! [`ReportRef`]: crate::ReportRef
20//! [`ReportCollection`]: crate::report_collection::ReportCollection
21//!
22//! # Ownership Markers
23//!
24//! Ownership markers control whether reports and report references can be
25//! mutated and cloned.
26//!
27//! For owned reports ([`Report<C, O, T>`](crate::Report)), the ownership marker
28//! `O` can be:
29//! - [`Mutable`]: Unique ownership - the report can be mutated but not cloned
30//! - [`Cloneable`]: Shared ownership - the report can be cloned but not mutated
31//!
32//! For report references ([`ReportRef<C, O>`](crate::ReportRef)), the ownership
33//! marker `O` can be:
34//! - [`Cloneable`]: Enables [`clone_arc`](crate::ReportRef::clone_arc) to get
35//!   an owned report
36//! - [`Uncloneable`]: Does not provide
37//!   [`clone_arc`](crate::ReportRef::clone_arc)
38//!
39//! # Thread Safety Markers
40//!
41//! Thread safety markers control whether reports can be sent between threads or
42//! shared across threads. These appear as the third type parameter (`T`) in
43//! [`Report<C, O, T>`](crate::Report):
44//!
45//! - [`SendSync`]: The report and all its contents are `Send + Sync`, allowing
46//!   the report to cross thread boundaries.
47//! - [`Local`]: The report contains non-thread-safe data (like `Rc` or raw
48//!   pointers) and cannot be sent between threads.
49//!
50//! # Examples
51//!
52//! ## Creating Reports with Different Ownership Semantics
53//!
54//! ```
55//! use rootcause::prelude::*;
56//!
57//! // Mutable report - can be modified by adding context and attachments
58//! let mut report: Report<String, markers::Mutable> = report!("Error".to_string());
59//! let report: Report<String, markers::Mutable> = report.attach("Additional context");
60//!
61//! // Convert to cloneable report - can be cloned but not mutated
62//! let cloneable: Report<String, markers::Cloneable> = report.into_cloneable();
63//! let cloned: Report<String, markers::Cloneable> = cloneable.clone();
64//! assert_eq!(format!("{}", cloneable), format!("{}", cloned));
65//! ```
66//!
67//! ## Working with Thread Safety
68//!
69//! ```
70//! use std::rc::Rc;
71//!
72//! use rootcause::prelude::*;
73//!
74//! // Thread-safe report with String (String is Send + Sync)
75//! let thread_safe: Report<String, markers::Mutable, markers::SendSync> =
76//!     report!("Thread-safe error".to_string());
77//!
78//! // Can be sent to another thread
79//! std::thread::spawn(move || {
80//!     println!("{}", thread_safe);
81//! });
82//!
83//! // Local report with Rc (Rc is !Send + !Sync)
84//! let local_data: Rc<String> = Rc::new("Not thread-safe".to_string());
85//! let local_report: Report<Rc<String>, markers::Mutable, markers::Local> = report!(local_data);
86//! // local_report cannot be sent to another thread - won't compile
87//! ```

93/// Marker type for owned reports with unique ownership.
94///
95/// This marker is used exclusively with [`Report<C, Mutable, T>`] (not
96/// [`ReportRef`]). It indicates that the report has unique ownership of its
97/// data, which allows mutation operations but prevents cloning.
98///
99/// [`Report<C, Mutable, T>`]: crate::Report
100/// [`ReportRef`]: crate::ReportRef
101///
102/// # Available Operations
103///
104/// With `Mutable` ownership, you can:
105/// - Add attachments with [`attach`](crate::Report::attach)
106/// - Add parent context with [`context`](crate::Report::context)
107/// - Get mutable access with [`as_mut`](crate::Report::as_mut)
108/// - Convert to [`Cloneable`] with
109///   [`into_cloneable`](crate::Report::into_cloneable)
110///
111/// # Examples
112///
113/// ```
114/// use rootcause::prelude::*;
115///
116/// let report: Report<String, markers::Mutable> = report!("Database error".to_string());
117///
118/// // Can add attachments (consumes and returns the report)
119/// let report: Report<String, markers::Mutable> = report.attach("connection timeout");
120///
121/// // Can add parent context
122/// let report: Report<String, markers::Mutable> =
123///     report.context("Failed to fetch user data".to_string());
124///
125/// // Cannot clone - Mutable reports don't implement Clone
126/// // let cloned = report.clone(); // ❌ Won't compile
127/// ```
128#[derive(Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Debug, Default, Hash)]
129pub struct Mutable;

131/// Marker type for cloneable reports and report references.
132///
133/// This marker is used with both [`Report<C, Cloneable, T>`](crate::Report) and
134/// [`ReportRef<C, Cloneable>`](crate::ReportRef). It indicates shared ownership
135/// using reference counting (`Arc` internally), which allows cheap cloning but
136/// prevents mutation.
137///
138/// # Usage with Report
139///
140/// For [`Report<C, Cloneable, T>`](crate::Report), this marker means the report
141/// itself implements `Clone`, allowing you to cheaply clone the entire report
142/// (shallow copy via `Arc`).
143///
144/// # Usage with ReportRef
145///
146/// For [`ReportRef<C, Cloneable>`](crate::ReportRef), the marker enables the
147/// [`clone_arc`](crate::ReportRef::clone_arc) method, which clones the
148/// underlying `Arc` to produce an owned [`Report<C, Cloneable,
149/// T>`](crate::Report). Note that `ReportRef` itself is always `Copy` and
150/// `Clone` regardless of the ownership marker - the `Cloneable`
151/// marker specifically enables converting the reference back to an owned
152/// report.
153///
154/// # When to Use
155///
156/// Use `Cloneable` reports when you need to:
157/// - Share an error report across multiple code paths
158/// - Store reports in collections that require `Clone`
159/// - Return the same error from multiple places without deep copying
160///
161/// # Converting to Cloneable
162///
163/// Convert a [`Mutable`] report to `Cloneable` using
164/// [`into_cloneable`](crate::Report::into_cloneable):
165///
166/// ```
167/// use rootcause::prelude::*;
168///
169/// let report: Report<String, markers::Mutable> = report!("Error".to_string());
170///
171/// // Convert to cloneable
172/// let cloneable: Report<String, markers::Cloneable> = report.into_cloneable();
173///
174/// // Now can clone cheaply (shallow clone via Arc)
175/// let clone1: Report<String, markers::Cloneable> = cloneable.clone();
176/// let clone2: Report<String, markers::Cloneable> = cloneable.clone();
177/// ```
178///
179/// # Examples
180///
181/// Cloning owned reports:
182///
183/// ```
184/// use rootcause::prelude::*;
185///
186/// fn process_error(error: Report<String, markers::Cloneable>) {
187///     // Can clone the error to pass to multiple handlers
188///     let for_logging = error.clone();
189///     let for_metrics = error.clone();
190///
191///     println!("Logging: {}", for_logging);
192///     println!("Metrics: {}", for_metrics);
193/// }
194///
195/// let report: Report<String> = report!("An error occurred".to_string());
196/// process_error(report.into_cloneable());
197/// ```
198///
199/// Using `clone_arc` on report references:
200///
201/// ```
202/// use rootcause::{ReportRef, prelude::*};
203///
204/// let report: Report<String, markers::Cloneable> = report!("Error".to_string()).into_cloneable();
205///
206/// // Get a reference (ReportRef is Copy, so this is cheap)
207/// let report_ref: ReportRef<String, markers::Cloneable> = report.as_ref();
208///
209/// // Clone the underlying Arc to get an owned Report
210/// let owned: Report<String, markers::Cloneable> = report_ref.clone_arc();
211/// ```
212#[derive(Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Debug, Default, Hash)]
213pub struct Cloneable;

215/// Marker type for non-cloneable report references.
216///
217/// This marker is used exclusively with [`ReportRef<C,
218/// Uncloneable>`](crate::ReportRef) (not [`Report`](crate::Report)). It
219/// indicates that the reference does not provide the
220/// [`clone_arc`](crate::ReportRef::clone_arc) method to obtain an owned report.
221///
222/// Note that `ReportRef` itself is always `Copy` and `Clone` - you can always
223/// copy the reference itself. The `Uncloneable` marker only prevents cloning
224/// the underlying `Arc` to get an owned `Report`.
225///
226/// # Common Uses
227///
228/// `Uncloneable` references typically arise in two situations:
229///
230/// 1. **Taking a reference to a `Mutable` report**: When you call
231///    [`as_ref`](crate::Report::as_ref) on a `Report<C, Mutable>`, you get a
232///    `ReportRef<C, Uncloneable>` because the underlying report has unique
233///    ownership.
234///
235/// 2. **Explicitly restricting cloneability**: You can convert a `ReportRef<C,
236///    Cloneable>` to `ReportRef<C, Uncloneable>` when you want to pass a
237///    reference that explicitly cannot use `clone_arc`, ensuring the recipient
238///    can only inspect the report without obtaining ownership. This can be
239///    useful in APIs that need to accept both cloneable and uncloneable
240///    references.
241///
242///
243/// # Examples
244///
245/// Taking a reference to a `Mutable` report:
246///
247/// ```
248/// use rootcause::{ReportRef, prelude::*};
249///
250/// let report: Report<String, markers::Mutable> = report!("An error occurred".to_string());
251///
252/// // Taking a reference to a Mutable report gives an Uncloneable reference
253/// let report_ref: ReportRef<String, markers::Uncloneable> = report.as_ref();
254///
255/// // The reference itself can be copied (ReportRef is Copy)
256/// let copy = report_ref;
257///
258/// // But you cannot clone the underlying Arc to get an owned Report
259/// // let owned = report_ref.clone_arc(); // ❌ Method not available
260/// ```
261///
262/// Explicitly converting to `Uncloneable`:
263///
264/// ```
265/// use rootcause::{ReportRef, prelude::*};
266///
267/// let report: Report<String, markers::Cloneable> = report!("Error".to_string()).into_cloneable();
268///
269/// let cloneable_ref: ReportRef<String, markers::Cloneable> = report.as_ref();
270///
271/// // Convert to uncloneable to restrict the recipient's ability to clone
272/// let uncloneable_ref: ReportRef<String, markers::Uncloneable> = cloneable_ref.into_uncloneable();
273/// ```
274#[derive(Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Debug, Default, Hash)]
275pub struct Uncloneable;

277/// Marker type indicating that a report and all its contents are `Send + Sync`.
278///
279/// This is the default thread-safety marker for reports. When all the context
280/// objects and attachments in a report implement `Send + Sync`, the report
281/// itself can be safely sent to other threads and shared between threads.
282///
283/// # When to Use
284///
285/// Most standard types in Rust are `Send + Sync`, including:
286/// - Primitive types (`i32`, `String`, `Vec`, etc.)
287/// - Most standard library types
288/// - Types explicitly designed for concurrent use
289///
290/// Use `SendSync` (the default) unless you have a specific need for
291/// thread-local data.
292///
293/// # Examples
294///
295/// ```
296/// use std::thread;
297///
298/// use rootcause::prelude::*;
299///
300/// // String is Send + Sync, so the report is too
301/// let report: Report<String, markers::Mutable, markers::SendSync> =
302///     report!("Thread-safe error".to_string());
303///
304/// // Can send to another thread
305/// thread::spawn(move || {
306///     println!("Error in thread: {}", report);
307/// })
308/// .join()
309/// .unwrap();
310/// ```
311#[derive(Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Debug, Default, Hash)]
312pub struct SendSync;
313
314/// Marker type indicating that a report is not `Send` or `Sync`.
315///
316/// This marker is used when a report contains thread-local data that cannot be
317/// safely sent between threads or shared across threads. Common examples
318/// include `Rc`, raw pointers, or types that explicitly opt out of
319/// `Send`/`Sync`.
320///
321/// # When to Use
322///
323/// Use `Local` when your error context or attachments contain:
324/// - `Rc<T>` or `Weak<T>` (use `Arc<T>` for thread-safe alternative)
325/// - Raw pointers (`*const T`, `*mut T`)
326/// - Types that wrap thread-local storage
327/// - Any type that is `!Send` or `!Sync`
328///
329/// # Converting to Local
330///
331/// You can convert a thread-safe report to a local one using
332/// [`into_local`](crate::Report::into_local), or create a local report directly
333/// when the context type is not `Send + Sync`.
334///
335/// # Examples
336///
337/// ```
338/// use std::rc::Rc;
339///
340/// use rootcause::prelude::*;
341///
342/// // Rc is not Send or Sync, so the report must be Local
343/// let local_data: Rc<String> = Rc::new("Thread-local error".to_string());
344/// let report: Report<Rc<String>, markers::Mutable, markers::Local> = report!(local_data);
345///
346/// // This report cannot be sent to another thread
347/// // std::thread::spawn(move || {
348/// //     println!("{}", report); // ❌ Won't compile
349/// // });
350/// ```
351///
352/// Converting a thread-safe report to local:
353///
354/// ```
355/// use std::rc::Rc;
356///
357/// use rootcause::prelude::*;
358///
359/// let report: Report<String> = report!("Error".to_string());
360///
361/// // Convert to local report so we can use thread-local data
362/// let local_report: Report<String, markers::Mutable, markers::Local> = report.into_local();
363/// ```
364#[derive(Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Debug, Default, Hash)]
365pub struct Local;

382/// Marker trait for ownership semantics of owned reports.
383///
384/// This trait is implemented for [`Mutable`] and [`Cloneable`], the two
385/// ownership markers that can be used with [`Report<C, O, T>`](crate::Report).
386/// It exists primarily to enable the associated type `RefMarker`, which
387/// determines the ownership marker used when creating a reference to the
388/// report.
389///
390/// # Relationship to Report Construction
391///
392/// While this trait defines the ownership modes, the actual enforcement of
393/// ownership invariants happens at report construction time. It is impossible
394/// to construct a `Report<_, Mutable>` that doesn't have unique ownership, or a
395/// `Report<_, Cloneable>` that hasn't been properly set up for shared
396/// ownership.
397///
398/// # Ownership Modes
399///
400/// - [`Mutable`]: Indicates unique ownership. The report can be mutated but not
401///   cloned. Methods like [`attach`](crate::Report::attach) and
402///   [`as_mut`](crate::Report::as_mut) are only available in this mode.
403///
404/// - [`Cloneable`]: Indicates shared ownership via reference counting. The
405///   report can be cloned cheaply but cannot be mutated since there may be
406///   multiple references.
407///
408/// # Associated Types
409///
410/// The `RefMarker` associated type determines what kind of reference you get
411/// when calling [`as_ref`](crate::Report::as_ref):
412/// - For [`Mutable`], this is [`Uncloneable`] (the reference cannot use
413///   `clone_arc`)
414/// - For [`Cloneable`], this is [`Cloneable`] (the reference can use
415///   `clone_arc`)
416///
417/// # Implementation
418///
419/// This trait is sealed and cannot be implemented outside of this crate. You
420/// should use the provided implementations for [`Mutable`] and [`Cloneable`].
421pub trait ReportOwnershipMarker: sealed_report_ownership_marker::Sealed {
422    /// The ownership marker for references to this report type.
423    ///
424    /// This determines what type of reference is created when you call
425    /// [`Report::as_ref`](crate::Report::as_ref):
426    ///
427    /// - For [`Mutable`]: Returns [`ReportRef<C,
428    ///   Uncloneable>`](crate::ReportRef) because the underlying report has
429    ///   unique ownership
430    /// - For [`Cloneable`]: Returns [`ReportRef<C,
431    ///   Cloneable>`](crate::ReportRef) because the underlying report already
432    ///   uses shared ownership
433    type RefMarker;
434}
435impl ReportOwnershipMarker for Mutable {
436    type RefMarker = Uncloneable;
437}
438impl ReportOwnershipMarker for Cloneable {
439    type RefMarker = Cloneable;
440}

442/// Marker trait combining object and thread-safety requirements.
443///
444/// This trait enforces thread-safety constraints for context and attachment
445/// data at report construction time. Reports can only be constructed when their
446/// context and attachments satisfy the requirements of the thread-safety
447/// marker.
448///
449/// # Implementations
450///
451/// - For `T = Local`: Implemented for all `Sized + 'static` types, regardless
452///   of their `Send`/`Sync` status. This allows using types like `Rc` in local
453///   reports.
454///
455/// - For `T = SendSync`: Implemented only for `Sized + 'static` types that are
456///   also `Send + Sync`. This ensures thread-safe reports can only be
457///   constructed with thread-safe data.
458///
459/// # Enforcement at Construction
460///
461/// The key insight is that this trait is used as a bound during report
462/// construction. You cannot create a `Report<C, _, SendSync>` unless `C:
463/// ObjectMarkerFor<SendSync>`, which requires `C: Send + Sync`. This makes it
464/// impossible to accidentally create an invalid report:
465///
466/// ```compile_fail
467/// use std::rc::Rc;
468/// use rootcause::prelude::*;
469///
470/// // This won't compile because Rc is not Send + Sync
471/// let rc_data: Rc<String> = Rc::new("error".to_string());
472/// let report: Report<Rc<String>, markers::Mutable, markers::SendSync> = report!(rc_data);
473/// ```
474///
475/// Use [`Local`] instead for non-thread-safe data:
476///
477/// ```
478/// use std::rc::Rc;
479///
480/// use rootcause::prelude::*;
481///
482/// let rc_data: Rc<String> = Rc::new("error".to_string());
483/// let report: Report<Rc<String>, markers::Mutable, markers::Local> = report!(rc_data);
484/// ```
485pub trait ObjectMarkerFor<T>: sealed_object_marker::Sealed + Sized + 'static {
486    /// Runs report creation hooks specific to this thread-safety marker.
487    #[doc(hidden)]
488    #[track_caller]
489    fn run_creation_hooks(report: ReportMut<'_, dyn Any, T>);
490}
491
492impl<O: Sized + 'static> ObjectMarkerFor<Local> for O {
493    #[inline(always)]
494    fn run_creation_hooks(report: ReportMut<'_, dyn Any, Local>) {
495        crate::hooks::report_creation::run_creation_hooks_local(report);
496    }
497}
498
499impl<O: Sized + 'static> ObjectMarkerFor<SendSync> for O
500where
501    O: Send + Sync,
502{
503    #[inline(always)]
504    fn run_creation_hooks(report: ReportMut<'_, dyn Any, SendSync>) {
505        crate::hooks::report_creation::run_creation_hooks_sendsync(report);
506    }
