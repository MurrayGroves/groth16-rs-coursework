23    /// A mutable reference to a [`Report`].
24    ///
25    /// [`ReportMut`] provides mutable access to a report's children and
26    /// attachments while maintaining safe borrowing semantics. Unlike owned
27    /// reports, mutable references cannot be consumed for chaining
28    /// operations like [`Report::context`] or [`Report::attach`].
29    ///
30    /// # Key Characteristics
31    ///
32    /// - **Not `Copy` or `Clone`**: Ensures exclusive mutable access
33    /// - **Lifetime-bound**: Tied to the lifetime of the underlying report
34    /// - **Two type parameters**: Has context type `C` and thread safety marker
35    ///   `T` (no ownership marker since mutable references are always uniquely
36    ///   owned)
37    ///
38    /// # Thread Safety
39    ///
40    /// Unlike [`Report`] and [`ReportRef`], you cannot change the thread safety
41    /// marker on [`ReportMut`]:
42    ///
43    /// - You cannot convert [`SendSync`] → [`Local`] because that would allow
44    ///   adding non-thread-safe data to a report that should remain thread-safe
45    /// - You cannot convert [`Local`] → [`SendSync`] because that would allow
46    ///   cloning a child report with thread-local data and sending it across
47    ///   threads
48    ///
49    /// # Common Usage
50    ///
51    /// ```
52    /// use rootcause::{ReportMut, prelude::*};
53    ///
54    /// let mut report: Report = report!("error message");
55    ///
56    /// // Get mutable access to modify children or attachments
57    /// let mut report_mut: ReportMut<'_> = report.as_mut();
58    /// report_mut
59    ///     .children_mut()
60    ///     .push(report!("child error").into_cloneable());
61    ///
62    /// println!("{}", report);
63    /// ```
64    ///
65    /// [`Report`]: crate::Report
66    /// [`Report::context`]: crate::Report::context
67    /// [`Report::attach`]: crate::Report::attach
68    /// [`ReportRef`]: crate::ReportRef
69    /// [`SendSync`]: crate::markers::SendSync
70    /// [`Local`]: crate::markers::Local
71    //
72    // # Safety invariants
73    //
74    // This reference behaves like a `&'a mut Report<C, Mutable, T>` for some unknown
75    // `C` and upholds the usual safety invariants of mutable references:
76    //
77    // 1. The pointee is properly initialized for the entire lifetime `'a`.
78    // 2. The pointee is not aliased for the entire lifetime `'a`.
79    // 3. Like a `&'a mut T`, it is possible to reborrow this reference to a shorter lifetime. The
80    //    borrow checker will ensure that original longer lifetime is not used while the shorter
81    //    lifetime exists.
82    #[repr(transparent)]
83    pub struct ReportMut<'a, Context: ?Sized + 'static = dyn Any, ThreadSafety: 'static = SendSync> {
84        /// # Safety
85        ///
86        /// The following safety invariants are guaranteed to be upheld as long
87        /// as this struct exists and must be continue to be upheld as
88        /// long as the inner `RawReportMut` exists:
89        ///
90        /// 1. `C` must either be a concrete type bounded by `Sized`, or `dyn
91        ///    Any`.
92        /// 2. `T` must either be `SendSync` or `Local`.
93        /// 3. If `C` is a concrete type: The context embedded in the report
94        ///    must be of type `C`
95        /// 4. The strong count of the underlying `triomphe::Arc` is exactly 1.
96        /// 5. All references to any sub-reports of this report are compatible
97        ///    with shared ownership. Specifically there are no references with
98        ///    an assumption that the strong_count is `1`.
99        /// 7. If `T = SendSync`: All contexts and attachments in the report and
100        ///    all sub-reports must be `Send+Sync`
101        /// 8. If `T = Local`: No other references to this report are allowed to
102        ///    have `T = SendSync`
103        raw: RawReportMut<'a>,
104        _context: PhantomData<Context>,
105        _thread_safety: PhantomData<ThreadSafety>,
106    }

597    /// Returns an iterator over child reports in the report hierarchy
598    /// (excluding this report).
599    ///
600    /// The iterator visits reports in a depth-first order: it first visits the
601    /// current report's children, then recursively visits each child report
602    /// and all of their descendants before moving to the next sibling.
603    /// Unlike [`ReportMut::iter_reports`], this method does NOT include the
604    /// report on which it was called - only its descendants.
605    ///
606    /// This method always returns cloneable report references, making it
607    /// suitable for scenarios where you need to store or pass around the
608    /// report references.

611    /// including the current one.

613    /// # Examples

621    /// // Build hierarchy using .context() which creates new nodes
622    /// let with_context1 = error1.context("context for error 1");

625    /// // Create root that contains both context nodes as children

630    ///     .push(with_context2.into_dyn_any().into_cloneable());
631    ///
632    /// let root_mut: ReportMut<'_, &'static str> = root.as_mut();
633    ///
634    /// let sub_reports: Vec<String> = root_mut
635    ///     .iter_sub_reports()
636    ///     .map(|report| report.format_current_context().to_string())
637    ///     .collect();
638    ///
639    /// // Current "root" report is NOT included in the results

1081        // 6. This is guaranteed by the invariants of this type.
1082        // 7. This is guaranteed by the invariants of this type.
1083        unsafe { ReportMut::from_raw(raw) }
