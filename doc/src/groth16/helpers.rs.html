5pub(crate) fn rand_scalar<T, S>(rng: &mut T) -> S
6where
7    T: Rng,
8    S: Field,
9{
10    let mut bytes = [0; 256];
11    let mut out = None;
12    while out.is_none() {
13        rng.fill_bytes(&mut bytes);
14        out = S::from_random_bytes(&bytes);
15    }
16
17    out.unwrap()
18}

20pub(crate) fn ark_se<S, A: CanonicalSerialize>(a: &A, s: S) -> Result<S::Ok, S::Error>
21where
22    S: serde::Serializer,
23{
24    // ark_se and ark_de taken from https://github.com/arkworks-rs/algebra/issues/178#issuecomment-1413219278
25    let mut bytes = vec![];
26    a.serialize_with_mode(&mut bytes, Compress::Yes)
27        .map_err(serde::ser::Error::custom)?;
28    s.serialize_bytes(&bytes)
29}
30
31pub(crate) fn ark_de<'de, D, A: CanonicalDeserialize>(data: D) -> Result<A, D::Error>
32where
33    D: serde::de::Deserializer<'de>,
34{
35    let s: Vec<u8> = serde::de::Deserialize::deserialize(data)?;
36    let a = A::deserialize_with_mode(s.as_slice(), Compress::Yes, Validate::Yes);
37    a.map_err(serde::de::Error::custom)
38}
